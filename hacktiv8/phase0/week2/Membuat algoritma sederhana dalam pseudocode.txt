// Tugas 3
SIMPAN “angka”
TENTUKAN “index” sama dengan 1
WHILE “index” kurang dari sama dengan “angka”:
	IF index habis dibagi 3:
		TAMPILKAN “index”
	ENDIF
	ULANGI “index”
ENDWHILE
